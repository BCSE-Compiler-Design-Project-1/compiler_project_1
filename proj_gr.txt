start :- void main ( ) compound_stmt
stmt :- compound_stmt | exp_stmt | read_stmts | write_stmts | sel_stmt
other :- compound_stmt | exp_stmt | read_stmt | write_stmt
stmts :-  stmts stmt | stmt
compound_stmt :- { } | { stmts } | { decls } | { decls stmts }
read_stmts :- read_stmts read_stmt | read_stmt
write_stmts :- write_stmts write_stmt | write_stmt
read_stmt :-  cin inputs
inputs :- INP_OP id inputs | INP_OP id ;
INP_OP :- >>
write_stmt :- cout outputs
outputs :- OUT_OP exp outputs | OUT_OP exp ;
OUT_OP :- <<
decls :- decls decl | decl
decl :- type list ;
type :- int | float | char
list :- list , id | id
exp_stmt :- ; | exp ;
sel_stmt :- IF ( exp ) THEN other optional_tail
optional_tail :- ELSE tail | sel_stmt | #
tail :- other | sel_stmt
exp :- assignment_exp | exp , assignment_exp
assignment_exp :- equality_exp = equality_exp | equality_exp
equality_exp :- equality_exp equality_OP rel_exp | rel_exp
equality_OP :- == | !=
rel_exp :- rel_exp rel_OP add_exp | add_exp
rel_OP :- < | <= | > | >=   
add_exp :- add_exp add_OP mul_exp | mul_exp
add_OP :- + | -
mul_exp :- mul_exp mul_OP pri_exp | pri_exp
mul_OP :- * | / | %
pri_exp :- ( exp ) | id | constant

    int id , id ; float id ; 
    IF ( exp ) THEN {
        cin >> id >> id ;
        IF ( exp ) THEN id = id + id ;
        ELSE IF ( exp ) THEN  id = id * id ;
        ELSE id = constant ;
    } ELSE IF ( exp ) THEN {
        cout << constant << id << ( id + ( id * id ) - id ) / constant ;
        IF ( exp ) THEN id = id - constant ;
        IF ( exp ) THEN cout << id ; 
        ELSE id = id % constant ;
    } ELSE id = id / constant ;
} $
