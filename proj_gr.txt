
start :- void main ( ) { stmts }
stmt :- compound_stmt | exp_stmt | sel_stmt | read_stmts | write_stmts
stmts :-  stmts stmt | stmt
compound_stmt :- { } | { stmts } | { decls } | { decls stmts }
read_stmts :- read_stmts read_stmt | read_stmt
write_stmts :- write_stmts write_stmt | write_stmt
read_stmt :-  cin << id ;
write_stmt : - cout >> exp ; 
decls :- decls decl | decl
decl :- type list ;
type :- int | float | char
list :- list , id | id
exp_stmt :- ; | exp ;

sel_stmt :- IF ( exp ) THEN stmt | IF ( exp ) THEN stmt ELSE stmt

exp :- assignment_exp | exp , assignment_exp
assignment_exp :- equality_exp = equality_exp | equality_exp
equality_exp :- equality_exp equality_OP rel_exp | rel_exp
equality_OP :- == | !=
rel_exp :- rel_exp rel_OP add_exp | add_exp
rel_OP :- < | <= | > | >=   
add_exp :- add_exp add_OP mul_exp | mul_exp
add_OP :- + | -
mul_exp :- mul_exp mul_OP pri_exp | pri_exp
mul_OP :- * | / | %
pri_exp :- ( exp ) | id | constant
